3Status::Statuses args_parse(int argc, char* argv[], ArgsVars* args_vars,
4 const Argument args_dict[], const int args_dict_len) {
5 assert(argv);
6 assert(argc);
7 assert(args_vars);
8
9 bool exit = false;
10 for (int i = 1; i < argc; i++) {
11 bool is_found = false;
12
13 for (int j = 0; j < args_dict_len; j++) {
14 if (strcmp(argv[i], args_dict[j].arg) == 0) {
15 switch (args_dict[j].func(args_dict, args_dict_len, &i, argc, argv, args_vars)) {
16 case ArgsMode::CONTINUE:
17 break;
18 case ArgsMode::EXIT:
19 exit = true;
20 break;
21 case ArgsMode::ERROR:
22 return Status::ARGS_ERROR;
23 default:
24 assert(0 && "args[j].func() returned wrong ArgsMode");
25 break;
26 }
27 is_found = true;
28 break;
29 }
30 }
31 if (is_found)
32 continue;
33
34 printf("Wrong console argument: %s\n"
35 "Use -h for help\n", argv[i]);
36 return Status::ARGS_ERROR;
37 }
38 if (exit)
39 return Status::OK_EXIT;
40
41 return Status::NORMAL_WORK;
42}
